\hline
{\em count} & number of 32-\/bit writes \\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}
Result number of successful writes; return value == min(count, 16) if completely successful 
\end{DoxyReturn}
\begin{DoxyNote}{Note}
At 48 MHz, it takes 114 usec to write 16 words 

This function is not interrupt safe. 
\end{DoxyNote}
